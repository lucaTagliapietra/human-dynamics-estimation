669 static bool parseRotationMatrix(const yarp::os::Value& ini, iDynTree::Rotation& rotation)
670 {
671  if (ini.isNull() || !ini.isList())
672  {
673  return false;
674  }
675  yarp::os::Bottle *outerList = ini.asList();
676  if (!outerList || outerList->size() != 3)
677  {
678  return false;
679  }
680  for (int row = 0; row < outerList->size(); ++row)
681  {
682  yarp::os::Value& innerValue = outerList->get(row);
683  if (innerValue.isNull() || !innerValue.isList())
684  {
685  return false;
686  }
687  yarp::os::Bottle *innerList = innerValue.asList();
688  if (!innerList || innerList->size() != 3)
689  {
690  return false;
691  }
692  for (int column = 0; column < innerList->size(); ++column)
693  {
694  rotation.setVal(row, column, innerList->get(column).asDouble());
695  }
696  }
697  return true;
698 }
