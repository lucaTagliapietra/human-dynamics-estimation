74  bool ReadOnlyRemoteControlBoard::open(Searchable& config) {
75  ConstString remote = config.find("remote").asString();
76  ConstString local = config.find("local").asString();
77 
78  if (local.empty()) {
79  yError("Problem connecting to remote controlboard, 'local' port prefix not given");
80  return false;
81  }
82 
83  if (remote.empty()) {
84  yError("Problem connecting to remote controlboard, 'remote' port name not given");
85  return false;
86  }
87 
88  ConstString carrier = config.check("carrier", Value("udp"), "default carrier for streaming robot state").asString();
89 
90  bool portProblem = false;
91  if (!m_extendedIntputStatePort.open(local + "/stateExt:i")) {
92  portProblem = true;
93  }
94 
95  if (!portProblem) {
96  m_extendedIntputStatePort.useCallback();
97  }
98 
99 
100  bool connectionProblem = false;
101  if (!portProblem) {
102  bool ok = false;
103  ok = Network::connect(remote + "/stateExt:o", m_extendedIntputStatePort.getName(), carrier);
104  if (!ok) {
105  connectionProblem = true;
106  yError("*** Extended port %s was not found on the controlBoardWrapper I'm connecting to.", (remote + "/stateExt:o").c_str());
107  }
108  }
109 
110  if (connectionProblem || portProblem) {
111  m_extendedIntputStatePort.close();
112  return false;
113  }
114 
115  Value &axesDescription = config.find("axesDescription");
116  if (axesDescription.isNull() || !axesDescription.isList()) {
117  yError("*** Option 'axesDescription' not found or malformed.");
118  m_extendedIntputStatePort.close();
119  return false;
120  }
121 
122  Bottle *axesDescriptionList = axesDescription.asList();
123  m_numberOfJoints = axesDescriptionList->size();
124  m_extendedIntputStatePort.init(m_numberOfJoints);
125  m_axes.reserve(m_numberOfJoints);
126 
127  for (int index = 0; index < m_numberOfJoints; ++index) {
128  const Value& axis = axesDescriptionList->get(index);
129  if (axis.isNull() || !axis.isList() || axis.asList()->size() != 2) {
130  yError("*** Option 'axesDescription' malformed at index %d.", index);
131  m_extendedIntputStatePort.close();
132  return false;
133  }
134  ConstString axisName = axis.asList()->get(0).asString();
135  JointTypeEnum axisVocab = static_cast<JointTypeEnum>(axis.asList()->get(1).asVocab());
136  m_axes.push_back(std::pair<yarp::os::ConstString, yarp::dev::JointTypeEnum>(axisName, axisVocab));
137  }
138 
139  return true;
140  }
