60  void HumanIKWorkerPool::runAndWait()
61  {
62  //Fill data for a thread
63  {
64  std::unique_lock<std::mutex> guard(m_inputMutex);
65  for (auto& linkPair : m_linkPairs) {
66  // Create a new struct of type WorkerData and pass it to the pool
67  WorkerTaskData taskData = {
68  linkPair,
69  m_segments[static_cast<size_t>(linkPair.parentFrameSegmentsIndex)],
70  m_segments[static_cast<size_t>(linkPair.childFrameSegmentsIndex)],
71  std::distance(&*(m_linkPairs.begin()), &linkPair) //this is not properly clear with the range-based iterators
72  };
73  m_tasks.push(taskData);
74  }
75  m_inputSynchronizer.notify_all();
76  }
77 
78  //as this call is blocking I have to wait for all the results

81 
82  m_results.clear();
83  m_outputSynchronizer.notify_all();
84  }

95  void HumanIKWorkerPool::computeJointVelocities(WorkerTaskData& task, iDynTree::VectorDynSize& relativeVelocity)
96  {
97  // Update kinDynComputations object
98  iDynTree::Vector3 worldGravity;
99  worldGravity.zero();
100  worldGravity(2) = -9.81;
101 
102  //Obtain the pointer to the kynDyn object, just as an alias
103  iDynTree::KinDynComputations *computations = task.pairInfo.kinDynComputations.get();
104  //As we computed the IK, set the state
105  task.pairInfo.jointVelocities.zero();
106  computations->setRobotState(task.pairInfo.jointConfigurations,
107  task.pairInfo.jointVelocities,
108  worldGravity);
109 
110  //Parent, child and relative jacobians
111  iDynTree::MatrixDynSize& relativeJacobian = task.pairInfo.relativeJacobian;
112  computations->getRelativeJacobian(task.pairInfo.parentFrameModelIndex,
113  task.pairInfo.childFrameModelIndex,
114  relativeJacobian);
115 
116  //Pseudo-invert the Jacobian.
117  //Compute the QR decomposition
118  task.pairInfo.jacobianDecomposition.compute(iDynTree::toEigen(relativeJacobian));
119  //the solve method on the decomposition directly solves the associated least-squares problem
120  relativeVelocity = task.childFrameInfo.velocities;
121  iDynTree::toEigen(relativeVelocity) -= iDynTree::toEigen(task.parentFrameInfo.velocities);
122  iDynTree::toEigen(task.pairInfo.jointVelocities) = task.pairInfo.jacobianDecomposition.solve(iDynTree::toEigen(relativeVelocity));
123  }

125  void HumanIKWorkerPool::worker() {
126  //Preallocate some thread-local variables to be used in the computation
127  iDynTree::VectorDynSize relativeVelocity(6);
128 
129  while (true) {
130 #ifdef EIGEN_RUNTIME_NO_MALLOC
131  Eigen::internal::set_is_malloc_allowed(false);
132 #endif
133 
134  std::unique_lock<std::mutex> guard(m_inputMutex);
135  m_inputSynchronizer.wait(guard, [&](){ return m_shouldTerminate || !m_tasks.empty(); } );
136  if (m_shouldTerminate) {
137  if (m_terminateCounter.empty()) {
138  yError("Trying to remove a thread which was not expected to exist");
139  } else {
140  m_terminateCounter.pop_back();
141  }
142  m_inputSynchronizer.notify_one();
143  break;
144  }
145 
146  WorkerTaskData task = m_tasks.front();
147  m_tasks.pop();
148  guard.unlock();
149 
150  //Do computations
151  //std::chrono::high_resolution_clock::time_point t1 = std::chrono::high_resolution_clock::now();
152  int ikResult = computeIK(task);
153  //std::cerr << "IK took " <<std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::high_resolution_clock::now() - t1).count() << "ms" << std::endl;
154 
155  if (ikResult < 0) {
156  yError("Failed to compute IK for frames %s, %s with error %d",
157  task.pairInfo.parentFrameName.c_str(),
158  task.pairInfo.childFrameName.c_str(),
159  ikResult);
160  }
161  computeJointVelocities(task, relativeVelocity);
162 
163  //Notify caller
164  std::unique_lock<std::mutex> outputGuard(m_outputMutex);
165  m_outputSynchronizer.wait(outputGuard, [&](){ return m_results.find(task.identifier) == m_results.end(); });
166 
167  //insert
168  m_results.insert(std::unordered_map<unsigned, int>::value_type(task.identifier, ikResult > 0));
169  m_outputSynchronizer.notify_all();
170 
171 #ifdef EIGEN_RUNTIME_NO_MALLOC
172  Eigen::internal::set_is_malloc_allowed(true);
173 #endif
174  }
175  }
