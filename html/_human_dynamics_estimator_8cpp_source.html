571  m_intermediateQuantities.covarianceDynamicsPriorInverseDecomposition.factorize(m_intermediateQuantities.covarianceDynamicsPriorInverse);
572 
573  // Expected value of the prior of the dynamics: E[p(d)], Eq. 10b
574  toEigen(m_intermediateQuantities.expectedDynamicsPriorRHS) = toEigen( m_priorDynamicsRegularizationCovarianceInverse) * toEigen(m_priorDynamicsRegularizationExpectedValue) - toEigen(m_dynamicsConstraintsMatrix).transpose() * toEigen(m_priorDynamicsConstraintsCovarianceInverse) * toEigen(m_dynamicsConstraintsBias);
575 
576  toEigen(m_intermediateQuantities.expectedDynamicsPrior) =
577  m_intermediateQuantities.covarianceDynamicsPriorInverseDecomposition.solve(toEigen(m_intermediateQuantities.expectedDynamicsPriorRHS));
578 
579  // Final result: covariance matrix of the whole-body dynamics, Eq. 11a
580  //TODO: find a way to map to iDynTree::SparseMatrix
581  m_covarianceDynamicsAPosterioriInverse = m_intermediateQuantities.covarianceDynamicsPriorInverse;
582  m_covarianceDynamicsAPosterioriInverse += toEigen(m_measurementsMatrix).transpose() * toEigen(m_priorMeasurementsCovarianceInverse) * toEigen(m_measurementsMatrix);
583 
584  // decompose m_covarianceDynamicsAPosterioriInverse
585  if (computePermutation) {
586 // m_intermediateQuantities.covarianceDynamicsAPosterioriInverseDecomposition.analyzePattern(toEigen(m_covarianceDynamicsAPosterioriInverse));
587  m_intermediateQuantities.covarianceDynamicsAPosterioriInverseDecomposition.analyzePattern(m_covarianceDynamicsAPosterioriInverse);
588  }
589 // m_intermediateQuantities.covarianceDynamicsAPosterioriInverseDecomposition.factorize(toEigen(m_covarianceDynamicsAPosterioriInverse));
590  m_intermediateQuantities.covarianceDynamicsAPosterioriInverseDecomposition.factorize(m_covarianceDynamicsAPosterioriInverse);
591 
592  // Final result: expected value of the whole-body dynamics, Eq. 11b
593  toEigen(m_intermediateQuantities.expectedDynamicsAPosterioriRHS) = (toEigen(m_measurementsMatrix).transpose() * toEigen(m_priorMeasurementsCovarianceInverse) * (toEigen(m_measurements) - toEigen(m_measurementsBias)) + m_intermediateQuantities.covarianceDynamicsPriorInverse * toEigen(m_intermediateQuantities.expectedDynamicsPrior));
594  toEigen(m_expectedDynamicsAPosteriori) =
595  m_intermediateQuantities.covarianceDynamicsAPosterioriInverseDecomposition.solve(toEigen(m_intermediateQuantities.expectedDynamicsAPosterioriRHS));
596 }

673 static bool parseMeasurementsPriorsOption(const yarp::os::Bottle& priorsGroup,
674  const iDynTree::BerdyHelper& berdy,
675  const std::string& optionPrefix,
676  iDynTree::SparseMatrix& parsedMatrix)
677 {
678  using std::pair;
679  using std::string;
680  //While this method may be applied for all the priors, let's stay focused on the measurements part.
681  /* cases are:
682  * 1) constant (for all elements)
683  * 2) different depending on the type of variable (see measurements)
684  * 3) in case of forces, contact/not in contact
685  *
686  * as ini file we can do the following:
687  * 1) easy: key (optionPrefix) => value. This can be a single value or the whole vector
688  * 2) use optionPrefix as prefix. Append the string representation of the variable type
689  * as before, this can be single value or whole (limited) vector
690  * 3) more complex. I would use the optionPrefix as before.
691  *
692  * example with measurements
693  * 1) cov_meas 1e-2
694  * 2) cov_meas_ACCELEROMETER (1e+1, 2e+1, 5)
695  * 3) cov_meas_NET_EXT_WRENCH ((specific_element LeftFoot)(1e+1, 2e+1, 5, 0.1, 0.1, 0.3))
696  *
697  * Option 1 needs to fill a simple Triplets with constant elements (of size - inputSize)
698  * Option 2 needs to know the type of considered variable and the corresponding offset-size in berdy
699  * Option 3 same as 2.
700  */
701 
702  assert(optionPrefix == "cov_measurements");
703  iDynTree::Triplets triplets;
704 
705  //Option 1
706  if (priorsGroup.check(optionPrefix, "Checking priors on " + optionPrefix)) {
707  yarp::os::Value& option = priorsGroup.find(optionPrefix);
708  if (!parseCovarianceMatrixOption(option, parsedMatrix.rows(), triplets)) {
709  yWarning("Malformed priors information for \"%s\"", optionPrefix.c_str());
710  } else {
711  }
712  }
713 
714  //Option 2: for each measurments type, fill specific stuff.
715  //This is where we lose genericity and we work only with measurements
716  /* This is the "tentative grammar" for the option
717  *
718  * option2: key value
719  * value: plain | constrained | '(' plain? constrained* ')'
720  * constrained: '((' "specific_element" string) plain ')'
721  * plain: positive_double | '(' positive_double+ ')'
722  */
723  const size_t berdySensorNumber = 7;
724  iDynTree::BerdySensorTypes sensorTypes[berdySensorNumber] = {
725  iDynTree::SIX_AXIS_FORCE_TORQUE_SENSOR,
726  iDynTree::ACCELEROMETER_SENSOR,
727  iDynTree::GYROSCOPE_SENSOR,
728  iDynTree::DOF_ACCELERATION_SENSOR,
729  iDynTree::DOF_TORQUE_SENSOR,
730  iDynTree::NET_EXT_WRENCH_SENSOR,
731  iDynTree::JOINT_WRENCH_SENSOR
732  };
733 
734  pair<size_t, string> sensorsInfo[berdySensorNumber] = {
735  pair<size_t, string>(6, "SIX_AXIS_FORCE_TORQUE_SENSOR"),
736  pair<size_t, string>(3, "ACCELEROMETER_SENSOR"),
737  pair<size_t, string>(3, "GYROSCOPE_SENSOR"),
738  pair<size_t, string>(1, "DOF_ACCELERATION_SENSOR"),
739  pair<size_t, string>(1, "DOF_TORQUE_SENSOR"),
740  pair<size_t, string>(6, "NET_EXT_WRENCH_SENSOR"),
741  pair<size_t, string>(6, "JOINT_WRENCH_SENSOR"),
742  };
743 
744 
745 
746  for (size_t i = 0; i < berdySensorNumber; ++i) {
747  const pair<size_t, string>& sensorInfo = sensorsInfo[i];
748 
749  if (priorsGroup.check(optionPrefix + "_" + sensorInfo.second, "Checking priors on " + optionPrefix)) {
750  yarp::os::Bottle& option = priorsGroup.findGroup(optionPrefix + "_" + sensorInfo.second);
751 
752  //Get the value
753  std::vector<pair<string, iDynTree::Triplets> > partialElements;
754  iDynTree::Triplets currentTriplets;
755 
756  //Try to parse it as if it were a plain type
757  if (!parseCovarianceMatrixOption(option.get(1), sensorInfo.first, currentTriplets)) {
758  // If not let's try more difficult parsing
759  yarp::os::Bottle *values = option.get(1).asList();
760  if (!values) {
761  //Empty value
762  continue;
763  }
764  for (int valueIndex = 0; valueIndex < values->size(); ++valueIndex) {
765  yarp::os::Value& currentValue = values->get(valueIndex);
766  if (valueIndex == 0) {
767  // This should parse the "plain" value which if present is always (and only) the first
768  currentTriplets.clear();
769  if (parseCovarianceMatrixOption(currentValue, sensorInfo.first, currentTriplets)) {
770  // plain value found. Go to next
771  partialElements.push_back(pair<string, iDynTree::Triplets>("", currentTriplets));
772  continue;
773  }
774  }
775  //If here we expect two elements
776  if (!currentValue.isList() || currentValue.asList()->size() != 2) {
777  continue;
778  }
779  yarp::os::Bottle *specificList = currentValue.asList();
780  //First element MUST be the name of the constrained link
781  if (!specificList->get(0).isList() || specificList->get(0).asList()->size() != 2
782  || specificList->get(0).asList()->get(0).asString() != "specific_element") {
783  continue;
784  }
785  std::string consideredFrame = specificList->get(0).asList()->get(1).asString();
786  //Second element can be a list or value. So parse it using the usual method
787  currentTriplets.clear();
788  if (!parseCovarianceMatrixOption(specificList->get(1), sensorInfo.first, currentTriplets)) {
789  continue;
790  }
791  partialElements.push_back(pair<string, iDynTree::Triplets>(consideredFrame, currentTriplets));
792  }
793 
794  } else {
795  partialElements.push_back(pair<string, iDynTree::Triplets>("", currentTriplets));
796  }
797 
798  if (partialElements.empty()) {
799  yWarning("Malformed priors information for \"%s\"", (optionPrefix + "_" + sensorInfo.second).c_str());
800  continue;
801  }
802 
803  //Now I have to set this triplets in the global one
804  for (auto &sensor : berdy.getSensorsOrdering()) {
805  for (pair<string, iDynTree::Triplets> element : partialElements) {
806  if (sensor.type != sensorTypes[i]) continue;
807  if (!element.first.empty() && sensor.id != element.first) continue;
808 
809  for (const auto &triplet : element.second) {
810  iDynTree::Triplet modifiedTriplet = triplet;
811  modifiedTriplet.row += sensor.range.offset;
812  modifiedTriplet.column += sensor.range.offset;
813  triplets.setTriplet(modifiedTriplet);
814  }
815  }
816  }
817  }
818  }
819 
820  if (!triplets.isEmpty()) {
821  parsedMatrix.setFromTriplets(triplets);
822  }
823  return true;
824 }
